Natural join for 1st query:
SELECT BOOK_ID,TITLE,PUBLISHER_NAME,AUTHOR_NAME,NO_OF_COPIES,PROGRAMME_ID FROM BOOK NATURAL JOIN BOOK_COPIES NATURAL JOIN BOOK_AUTHORS;

inner join for 1st query:

SELECT B.BOOK_ID, TITLE, PUBLISHER_NAME, AUTHOR_NAME,
BC.NO_OF_COPIES, BC.PROGRAMME_ID FROM 
BOOK B INNER JOIN BOOK_AUTHORS BA ON B.BOOK_ID=BA.BOOK_ID 
INNER JOIN BOOK_COPIES BC ON 
BA.BOOK_ID=BC.BOOK_ID;
